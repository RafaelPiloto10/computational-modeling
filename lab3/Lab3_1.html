import numpy as np

def forward_diff(x, y):
  dy = (y[1:] - y[:-1])
  dx = (x[1:] - x[:-1])
  return dy / dx

def smooth(x, y, window_size):
  i = 0
  z = []
  x_new = []

  while i < len(y) - window_size + 1:
      wa = np.sum(y[i:i+window_size]) / window_size
      
      z.append(wa)
      x_new.append(x[i + window_size//2])
        
      i += 1

  return x_new, z
